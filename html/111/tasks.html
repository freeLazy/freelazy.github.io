为什么不建议使用 += 进行字符串的拼接? 

// 1
string html = "<p>";
html += "hello";
html += ",";
html += "world";
html += "</p>"
Console.WriteLine(html);

// 2
string s = "";
for (int i = 0; i < 10; i++)
{
    s += i;
}
Console.WriteLine(s);

// 2
StringBuilder sb = new StringBuilder();
for (int i = 0; i < 10; i++)
{
    sb.Append(i);
}
Console.WriteLine(sb.ToString());
